4. DATABASE IMPLEMENTATION
Complete Database Schema:
sql
-- Users Table
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(255),
    email VARCHAR(255),
    pwd VARCHAR(255) NOT NULL,
    role VARCHAR(50) DEFAULT 'CA',
    is_admin TINYINT DEFAULT 0,
    birthdate DATE,
    profile_picture VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Client Masters Table
CREATE TABLE client_masters (
    id INT PRIMARY KEY AUTO_INCREMENT,
    uid INT NOT NULL,
    username VARCHAR(255) UNIQUE,
    name VARCHAR(255) NOT NULL,
    phone_no VARCHAR(20),
    email VARCHAR(255),
    pwd VARCHAR(255),
    status TINYINT DEFAULT 1,
    dob DATE,
    business_name VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (uid) REFERENCES users(id) ON DELETE CASCADE
);

-- Financial Year Master
CREATE TABLE fy_master (
    id INT PRIMARY KEY AUTO_INCREMENT,
    year VARCHAR(10) UNIQUE NOT NULL,
    status TINYINT DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Data/Document Table
CREATE TABLE data (
    id INT PRIMARY KEY AUTO_INCREMENT,
    uid INT NOT NULL,
    client_id INT NOT NULL,
    fy_id INT NOT NULL,
    file_title VARCHAR(255) NOT NULL,
    file_type VARCHAR(50),
    status VARCHAR(50) DEFAULT 'uploaded',
    gdrive_file_id VARCHAR(255),
    gdrive_web_link VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (uid) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (client_id) REFERENCES client_masters(id) ON DELETE CASCADE,
    FOREIGN KEY (fy_id) REFERENCES fy_master(id) ON DELETE CASCADE
);

-- Financial Year Data/Status Tracker
CREATE TABLE fy_data (
    id INT PRIMARY KEY AUTO_INCREMENT,
    uid INT NOT NULL,
    fy_client_id INT NOT NULL,
    fy_id INT NOT NULL,
    status INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (uid) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (fy_client_id) REFERENCES client_masters(id) ON DELETE CASCADE,
    FOREIGN KEY (fy_id) REFERENCES fy_master(id) ON DELETE CASCADE,
    UNIQUE KEY unique_client_fy (fy_client_id, fy_id)
);

-- Fees Master Table
CREATE TABLE fees_master (
    id INT PRIMARY KEY AUTO_INCREMENT,
    uid INT NOT NULL,
    cate INT NOT NULL,
    category_name VARCHAR(255),
    fees DECIMAL(10,2) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (uid) REFERENCES users(id) ON DELETE CASCADE,
    UNIQUE KEY unique_ca_category (uid, cate)
);
